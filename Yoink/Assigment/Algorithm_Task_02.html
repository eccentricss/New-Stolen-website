ðŸ‡¬ðŸ‡§ Advanced Dijkstra with Path Reconstruction


import heapq
import matplotlib.pyplot as plt
import networkx as nx

class AdvancedDijkstra:
    def __init__(self, graph):
        self.graph = graph
        self.distances = {}
        self.previous = {}
    
    def find_shortest_path(self, start, target):
        # Initialize distances
        self.distances = {node: float('infinity') for node in self.graph}
        self.distances[start] = 0
        self.previous = {node: None for node in self.graph}
        
        priority_queue = [(0, start)]
        
        while priority_queue:
            current_distance, current_node = heapq.heappop(priority_queue)
            
            # Skip if we found a better path already
            if current_distance > self.distances[current_node]:
                continue
            
            for neighbor, weight in self.graph[current_node].items():
                distance = current_distance + weight
                
                # Only consider this path if it's better
                if distance < self.distances[neighbor]:
                    self.distances[neighbor] = distance
                    self.previous[neighbor] = current_node
                    heapq.heappush(priority_queue, (distance, neighbor))
        
        return self._reconstruct_path(target)
    
    def _reconstruct_path(self, target):
        """Reconstruct the shortest path to target"""
        path = []
        current = target
        
        while current is not None:
            path.append(current)
            current = self.previous[current]
        
        return path[::-1]  # Return reversed path
    
    def visualize_graph(self, start, target):
        """Create visual representation of the graph and shortest path"""
        G = nx.DiGraph()
        
        # Add edges to graph
        for node in self.graph:
            for neighbor, weight in self.graph[node].items():
                G.add_edge(node, neighbor, weight=weight)
        
        # Create visualization
        pos = nx.spring_layout(G)
        plt.figure(figsize=(12, 8))
        
        # Draw all nodes and edges
        nx.draw_networkx_nodes(G, pos, node_color='lightblue', node_size=500)
        nx.draw_networkx_edges(G, pos, edge_color='gray', arrows=True)
        nx.draw_networkx_labels(G, pos)
        
        # Highlight shortest path
        shortest_path = self.find_shortest_path(start, target)
        path_edges = [(shortest_path[i], shortest_path[i+1]) for i in range(len(shortest_path)-1)]
        
        nx.draw_networkx_edges(G, pos, edgelist=path_edges, 
                              edge_color='red', width=2, arrows=True)
        nx.draw_networkx_nodes(G, pos, nodelist=shortest_path, 
                              node_color='red', node_size=700)
        
        # Add edge labels
        edge_labels = nx.get_edge_attributes(G, 'weight')
        nx.draw_networkx_edge_labels(G, pos, edge_labels=edge_labels)
        
        plt.title(f"Shortest Path from {start} to {target}")
        plt.axis('off')
        plt.show()
        
        return shortest_path

# Example usage with network similar to your dashboard
def demonstrate_algorithm():
    # Network topology similar to server infrastructure
    network_graph = {
        'WebServer': {'Database': 5, 'Cache': 2, 'LoadBalancer': 1},
        'Database': {'Backup': 8, 'WebServer': 5},
        'Cache': {'CDN': 3, 'WebServer': 2},
        'LoadBalancer': {'WebServer': 1, 'Cache': 2},
        'CDN': {'Cache': 3, 'User': 1},
        'Backup': {'Database': 8},
        'User': {'CDN': 1}
    }
    
    dijkstra = AdvancedDijkstra(network_graph)
    
    # Find shortest path from User to Backup
    path = dijkstra.find_shortest_path('User', 'Backup')
    print(f"Shortest path: {' -> '.join(path)}")
    print(f"Total distance: {dijkstra.distances['Backup']}ms")
    
    return dijkstra.visualize_graph('User', 'Backup')

# Run demonstration
if __name__ == "__main__":
    demonstrate_algorithm()
