import requests
import json

class WeatherApp:
    def __init__(self, api_key):
        self.api_key = api_key
        self.base_url = "http://api.openweathermap.org/data/2.5/weather"
    
    def get_weather(self, city_name, country_code=None):
        """Fetch current weather for specified city"""
        try:
            # Build query parameters
            query = f"{city_name},{country_code}" if country_code else city_name
            params = {
                'q': query,
                'appid': self.api_key,
                'units': 'metric',  # Celsius
                'lang': 'en'
            }
            
            # Make API request
            response = requests.get(self.base_url, params=params, timeout=10)
            response.raise_for_status()
            
            return self._parse_weather_data(response.json())
            
        except requests.exceptions.RequestException as e:
            return f"âŒ Network error: {e}"
        except ValueError as e:
            return f"âŒ Data parsing error: {e}"
    
    def _parse_weather_data(self, data):
        """Extract and format weather information"""
        main_info = data['main']
        weather_info = data['weather'][0]
        
        return {
            'city': data['name'],
            'country': data['sys']['country'],
            'temperature': main_info['temp'],
            'feels_like': main_info['feels_like'],
            'humidity': main_info['humidity'],
            'description': weather_info['description'].title(),
            'pressure': main_info['pressure'],
            'wind_speed': data['wind']['speed'] if 'wind' in data else 'N/A'
        }
    
    def display_weather(self, weather_data):
        """Format and display weather information"""
        if isinstance(weather_data, str):
            print(weather_data)
            return
        
        print(f"\n{'='*50}")
        print(f"ğŸŒ¤ï¸  WEATHER REPORT - {weather_data['city']}, {weather_data['country']}")
        print(f"{'='*50}")
        print(f"ğŸŒ¡ï¸  Temperature: {weather_data['temperature']}Â°C")
        print(f"ğŸ¤” Feels like: {weather_data['feels_like']}Â°C")

        print(f"ğŸ“ Conditions: {weather_data['description']}")
        print(f"ğŸŒ¬ï¸  Wind Speed: {weather_data['wind_speed']} m/s")
        print(f"ğŸ“Š Pressure: {weather_data['pressure']} hPa")
        print(f"{'='*50}")
